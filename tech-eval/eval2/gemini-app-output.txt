4. What are the most common page navigation patterns?

Python
import pandas as pd

# Load the events data
events = pd.read_csv('events.csv')

# Ensure timestamp is datetime
events['timestamp'] = pd.to_datetime(events['timestamp'])

# Sort by session and timestamp
events = events.sort_values(['sessionId', 'timestamp'])

# Create sequences of 'page' events for each session
patterns = events.groupby('sessionId')['page'].apply(lambda x: ' -> '.join(x)).reset_index()

# Count the frequency of each pattern
pattern_counts = patterns['page'].value_counts().reset_index()
pattern_counts.columns = ['pattern', 'count']
total_sessions = len(patterns)
pattern_counts['percentage'] = (pattern_counts['count'] / total_sessions) * 100

print("Top Page Navigation Patterns:")
print(pattern_counts.head(10))

# Also look at category transitions for page_view events
page_views = events[events['page'] == 'page_view']
cat_patterns = page_views.groupby('sessionId')['category'].apply(lambda x: ' -> '.join(x.dropna())).reset_index()
cat_pattern_counts = cat_patterns['category'].value_counts().reset_index()
cat_pattern_counts.columns = ['pattern', 'count']
print("\nTop Category Transitions:")
print(cat_pattern_counts.head(5))

# Check specific numbers for the response
top_pattern = pattern_counts.iloc[0]['pattern']
top_count = pattern_counts.iloc[0]['count']
top_pct = pattern_counts.iloc[0]['percentage']

second_pattern = pattern_counts.iloc[1]['pattern']
second_count = pattern_counts.iloc[1]['count']
second_pct = pattern_counts.iloc[1]['percentage']

third_pattern = pattern_counts.iloc[2]['pattern']
third_count = pattern_counts.iloc[2]['count']
third_pct = pattern_counts.iloc[2]['percentage']

print(f"\nTop 1: {top_pattern} ({top_count}, {top_pct:.2f}%)")
print(f"Top 2: {second_pattern} ({second_count}, {second_pct:.2f}%)")
print(f"Top 3: {third_pattern} ({third_count}, {third_pct:.2f}%)")

30. What is the funnel from Ticket created to Ticket assigned to level 1 support to WIP - level 1 support to Ticket solved by level 1 support to Ticket closed?

import pandas as pd

# Load the events data
events = pd.read_csv('events.csv')

# Convert timestamp to datetime

# Define the funnel stages
funnel_stages = [
    'Ticket created',
    'Ticket assigned to level 1 support',
    'WIP - level 1 support',
    'Ticket solved by level 1 support',
    'Ticket closed'
]

# Function to calculate funnel
def calculate_funnel(df, stages):
    sessions = df['sessionId'].unique()
    funnel_counts = []
    
    current_sessions = set(sessions)
    
    # Step 1: Ticket created
    step1_sessions = df[df['event'] == stages[0]]['sessionId'].unique()
    current_sessions = set(step1_sessions)
    funnel_counts.append(len(current_sessions))
    
    # Track the latest timestamp for the valid sessions
    # For simplicity in many process mining contexts, we just look for existence in order
    # but let's be more precise: the events must happen in order.
    
    # Let's get the first occurrence of the first event for each session
    session_timestamps = df[df['event'] == stages[0]].groupby('sessionId')['timestamp'].min().to_dict()
    
    for i in range(1, len(stages)):
        next_event = stages[i]
        valid_next_sessions = []
        new_timestamps = {}
        
        # Check sessions that were valid in previous step
        # They must have the next event at a timestamp > previous event's timestamp
        potential_events = df[(df['sessionId'].isin(current_sessions)) & (df['event'] == next_event)]
        
        for sess_id in current_sessions:
            prev_ts = session_timestamps.get(sess_id)
            # Find events for this session that happened after prev_ts
            matches = potential_events[(potential_events['sessionId'] == sess_id) & (potential_events['timestamp'] >= prev_ts)]
            if not matches.empty:
                valid_next_sessions.append(sess_id)
                new_timestamps[sess_id] = matches['timestamp'].min()
        
        current_sessions = set(valid_next_sessions)
        session_timestamps = new_timestamps
        funnel_counts.append(len(current_sessions))
        
    return funnel_counts

funnel_results = calculate_funnel(events, funnel_stages)

35. For tickets with resolution time under 12 hours, what is the funnel from Ticket created to Ticket assigned to level 1 support to Ticket solved by level 1 support to Ticket closed?

Python
import pandas as pd

# Load the data
events = pd.read_csv('events.csv')
attrs = pd.read_csv('record-attributes.csv')

# Filter sessions with resolutionHours < 12
fast_sessions = attrs[attrs['resolutionHours'] < 12]['sessionId'].unique()

# Filter events for these sessions
fast_events = events[events['sessionId'].isin(fast_sessions)]

# Funnel stages
stages = [
    'Ticket created',
    'Ticket assigned to level 1 support',
    'Ticket solved by level 1 support',
    'Ticket closed'
]

# Calculate funnel counts
# We want to see how many sessions reached each stage.
# Note: A funnel typically implies a sequence. 
# We'll check the number of unique sessions that contain each event, 
# but only considering sessions that had the previous event (though in reality, they likely all have 'created').

funnel_results = {}
current_sessions = set(fast_sessions)

for stage in stages:
    # Find sessions in the current set that have this specific event
    matching_sessions = fast_events[
        (fast_events['sessionId'].isin(current_sessions)) & 
        (fast_events['event'] == stage)
    ]['sessionId'].unique()
    
    funnel_results[stage] = len(matching_sessions)
    # Update current_sessions to only include those that reached this stage for a strict funnel
    # (Though for 'Ticket created', it's the base)
    current_sessions = set(matching_sessions)

print(f"Total sessions < 12h: {len(fast_sessions)}")
print("Funnel counts:", funnel_results)

# Let's also get some percentages
total = len(fast_sessions)
prev_count = total
for stage, count in funnel_results.items():
    drop_pct = (1 - count / prev_count) * 100 if prev_count > 0 else 100
    overall_pct = (count / total) * 100 if total > 0 else 0
    print(f"{stage}: {count} ({overall_pct:.2f}% of total, {drop_pct:.2f}% drop from prev)")
    prev_count = count
